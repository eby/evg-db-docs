== View Definition

[source,sql]
----
 SELECT DISTINCT l.hold,
    l.circ_lib,
    l.count
   FROM action.unfulfilled_hold_loops l
     JOIN action.unfulfilled_hold_min_loop m USING (hold)
  WHERE l.count = m.min;
----

== View Definition

[source,sql]
----
 SELECT mde.source,
    cdfm.name,
    cdfm.multi,
    cmf.label,
    cmf.id AS field,
    mde.value
   FROM metabib.display_entry mde
     JOIN config.metabib_field cmf ON cmf.id = mde.field
     JOIN config.display_field_map cdfm ON cdfm.field = mde.field;
----

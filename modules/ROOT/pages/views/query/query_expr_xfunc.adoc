== View Definition

[source,sql]
----
 SELECT expression.id,
    expression.parenthesize,
    expression.parent_expr,
    expression.seq_no,
    expression.column_name,
    expression.function_id,
    expression.negate
   FROM query.expression
  WHERE expression.type = 'xfunc'::text;
----

== View Definition

[source,sql]
----
 SELECT p.id,
    p.xact,
    p.payment_ts,
    p.voided,
    p.amount,
    p.note,
    p.amount_collected,
    p.accepting_usr,
    p.cash_drawer,
    c.relname AS payment_type
   FROM money.bnm_desk_payment p
     JOIN pg_class c ON p.tableoid = c.oid;
----

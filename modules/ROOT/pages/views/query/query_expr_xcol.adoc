== View Definition

[source,sql]
----
 SELECT expression.id,
    expression.parenthesize,
    expression.parent_expr,
    expression.seq_no,
    expression.table_alias,
    expression.column_name,
    expression.negate
   FROM query.expression
  WHERE expression.type = 'xcol'::text;
----

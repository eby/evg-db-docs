== View Definition

[source,sql]
----
 SELECT billing.id,
    billing.xact,
    billing.billing_ts,
    billing.voided,
    billing.voider,
    billing.void_time,
    billing.amount,
    billing.billing_type,
    billing.btype,
    billing.note,
    billing.create_date,
    billing.period_start,
    billing.period_end
   FROM money.billing
UNION ALL
 SELECT aged_billing.id,

:description: View reference for action.open_non_cataloged_circulation

== View Definition

[source,sql]
----
 SELECT ncc.id,
    ncc.patron,
    ncc.staff,
    ncc.circ_lib,
    ncc.item_type,
    ncc.circ_time
   FROM action.non_cataloged_circulation ncc
     JOIN config.non_cataloged_type nct ON nct.id = ncc.item_type
  WHERE (ncc.circ_time + nct.circ_duration) > CURRENT_TIMESTAMP;
----

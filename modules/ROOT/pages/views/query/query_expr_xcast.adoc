== View Definition

[source,sql]
----
 SELECT expression.id,
    expression.parenthesize,
    expression.parent_expr,
    expression.seq_no,
    expression.left_operand,
    expression.cast_type,
    expression.negate
   FROM query.expression
  WHERE expression.type = 'xcast'::text;
----

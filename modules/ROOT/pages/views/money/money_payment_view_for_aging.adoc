== View Definition

[source,sql]
----
 SELECT p.id,
    p.xact,
    p.payment_ts,
    p.voided,
    p.amount,
    p.note,
    p.payment_type,
    bnm.accepting_usr,
    bnmd.cash_drawer,
    maa.billing
   FROM money.payment_view p
     LEFT JOIN money.bnm_payment bnm ON bnm.id = p.id
     LEFT JOIN money.bnm_desk_payment bnmd ON bnmd.id = p.id
     LEFT JOIN money.account_adjustment maa ON maa.id = p.id;
----

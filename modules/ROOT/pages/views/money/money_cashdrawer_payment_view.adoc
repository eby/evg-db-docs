== View Definition

[source,sql]
----
 SELECT ou.id AS org_unit,
    ws.id AS cashdrawer,
    t.payment_type,
    p.payment_ts,
    p.amount,
    p.voided,
    p.note
   FROM actor.org_unit ou
     JOIN actor.workstation ws ON ou.id = ws.owning_lib
     LEFT JOIN money.bnm_desk_payment p ON ws.id = p.cash_drawer
     LEFT JOIN money.payment_view t ON p.id = t.id;
----

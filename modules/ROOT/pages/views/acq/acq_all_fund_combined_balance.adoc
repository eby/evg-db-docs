== View Definition

[source,sql]
----
 SELECT a.fund,
    a.amount - COALESCE(c.amount, 0::numeric) AS amount
   FROM acq.all_fund_allocation_total a
     LEFT JOIN ( SELECT fund_debit.fund,
            sum(fund_debit.amount) AS amount
           FROM acq.fund_debit
          GROUP BY fund_debit.fund) c USING (fund);
----
